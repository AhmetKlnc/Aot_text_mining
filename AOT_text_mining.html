def cleantext(text):
    text = " " + text + " "
    text = re.sub('', '', text) # parantez iindeki metni kaldrma
    text = re.sub('\w*\d\w*', '', text) # saylar kaldrma
    text = re.sub('\s+', ' ', text) # 1'den fazla boslugu sadce 1 bosluk yapmak icin
    text = re.sub('\n', ' ', text) # yeni bir satr geldiyse onu boluk yapmak icin
    text = re.sub('\"+', '', text) # herhangi bir alintiyi kaldirmak icin
    text = re.sub('[%s]' % re.escape(string.punctuation), '', text) # Tm noktalama isaretlerini silmek icin
    text = re.sub(r'[^\w\s]', '',text) # Diger noktalam isaretleri icin
    
    return text
